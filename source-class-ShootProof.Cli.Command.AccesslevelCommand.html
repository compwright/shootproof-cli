 65:      * @return array

 67:     protected function getValidators()

 84:                 // Require a password for certain access levels
 85:                 switch ($settings['accessLevel'])
 86:                 {
 87:                     case 'public_password':
 88:                     case 'private_password':
 89:                         // Was the --password option passed with no value?
 90:                         if ($value === true) {
 91:                             return false;
 92:                         }
 93: 
 94:                         $validator = new RequiredValidator();
 95:                         $settings[$setting] = $value;
 96:                         return $validator($value, $setting, $settings);
 97:                 }
 98: 
 99:                 return true;
100:             }),
101:         ];
102:     }

104:     /**
105:      * Changes the access level for a ShootProof gallery (event)
106:      *
107:      * If no event option is specified and a .shootproof file exists in the
108:      * directory, then event will be read from that file.
109:      *
110:      * @param string $dir The directory to process
111:      * @param Options $baseOptions
112:      * @param OptionsFactory $optionsFactory
113:      */
114:     protected function processDirectory($dir, Options $baseOptions, OptionsFactory $optionsFactory)
115:     {
116:         // Reload the options and read the directory config file
117:         $options = $optionsFactory->newInstance([], $this->getValidators());
118:         $configPath = new TildeExpander($dir) . '/.shootproof';
119:         $configLoader = new DotenvLoader($configPath);
120:         try {
121:             $configData = $configLoader->parse()->toArray();
122:             $options->loadOptionData($configData, false); // don't overwrite CLI data
123:             $this->logger->addDebug('ShootProof settings file found', [$configPath, $configData]);
124:         } catch (\InvalidArgumentException $e) {
125:         // ignore
126:             $this->logger->addDebug('ShootProof settings file not found', [$configPath]);
127:         }
128: 
129:         // Make sure all required options are present
130:         $options->validateAllRequired();
131:         $options->validate('password', $options->password);
132: 
133:         // Set the event access level
134:         $this->logger->addNotice('Setting access level', [
135:             'event' => $options->event,
136:             'level' => $options->accessLevel,
137:             'password' => $options->password ? str_repeat('*', strlen($options->password)) : null,
138:         ]);
139:         if (! $baseOptions->preview) {
140:             $result = $this->api->setEventAccessLevel($options->event, $options->accessLevel, $options->password);
141:             $this->logger->addDebug('Operation completed', $result);
142:         }
143:     }
144: }
