 45:     /**
 46:      * Registers a new ErrorHandler for a given Logger
 47:      *
 48:      * By default it will handle errors, exceptions and fatal errors
 49:      *
 50:      * @param LoggerInterface $logger
 51:      * @param array|false $errorLevelMap an array of E_* constant to LogLevel::* constant mapping,
 52:      *     or false to disable error handling
 53:      * @param int|false $exceptionLevel a LogLevel::* constant, or false to disable exception handling
 54:      * @param int|false $fatalLevel a LogLevel::* constant, or false to disable fatal error handling
 55:      * @return ErrorHandler
 56:      */
 57:     public static function register(
 58:         LoggerInterface $logger,
 59:         $errorLevelMap = array(),
 60:         $exceptionLevel = null,
 61:         $fatalLevel = null
 62:     ) {
 63:         // Use our own exception handler, but continue to use Monolog's error and fatal handlers
 64:         $monolog = MonologErrorHandler::register($logger, $errorLevelMap, false, $fatalLevel);
 65:         $handler = new static($logger, $monolog);
 66: 
 67:         if ($exceptionLevel !== false) {
 68:             // Our own exception handler
 69:             $handler->registerExceptionHandler($exceptionLevel);
 70:         }
 71: 
 72:         return $handler;
 73:     }
 74: 
 75:     /**
 76:      * @param LoggerInterface $logger A logger for capturing and writing error messages
 77:      * @param MonologErrorHandler $monolog Monolog error handler for passing through to Monolog
 78:      */

 80:     {
 81:         $this->logger = $logger;
 82:         $this->monologHandler = $monolog;
 83:     }

 86:      * Registers an exception handler
 87:      *
 88:      * @param string $level One of the LogLevel::* constants
 89:      * @param boolean $callPrevious Whether to also call previous exceptions in the chain
 90:      */
 91:     public function registerExceptionHandler($level = null, $callPrevious = true)
 92:     {
 93:         $prev = set_exception_handler(array($this, 'handleException'));
 94:         $this->uncaughtExceptionLevel = $level;
 95:         if ($callPrevious && $prev) {
 96:             $this->previousExceptionHandler = $prev;
 97:         }
 98:     }

105:     public function handleException(\Exception $e)
106:     {
107:         if ($e instanceof ValidatorException) {

109:                 $this->uncaughtExceptionLevel === null ? LogLevel::ERROR : $this->uncaughtExceptionLevel,
110:                 $e->getMessage(),

112:             );
113:         } else {

115:         }
