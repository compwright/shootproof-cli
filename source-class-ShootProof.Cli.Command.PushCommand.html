130:     /**
131:      * Pushes (uploads) files to ShootProof from the specified local directory
132:      *
133:      * @param string $dir The directory to process
134:      * @param Options $baseOptions
135:      * @param OptionsFactory $optionsFactory
136:      */
137:     protected function processDirectory($dir, Options $baseOptions, OptionsFactory $optionsFactory)
138:     {
139:         // Reload the options and read the directory config file
140:         $options = $optionsFactory->newInstance([], $this->getValidators(), $this->getDefaults());
141:         $configPath = new TildeExpander($dir) . '/.shootproof';
142:         $configLoader = new DotenvLoader($configPath);
143:         try {
144:             $configData = $configLoader->parse()->toArray();
145:             $options->loadOptionData($configData, false); // don't overwrite CLI data
146:             $this->logger->addDebug('Config file found', [$configPath, $configData]);
147:         } catch (\InvalidArgumentException $e) {
148:         // ignore
149:             $this->logger->addDebug('Config file not found', [$configPath]);
150:         }
151: 
152:         // Make sure all required options are present
153:         $options->validateAllRequired();
154: 
155:         // Get local file list
156:         $localFiles = array_map('basename', $this->getFileList($dir));
157: 
158:         $eventId = $options->event ? $options->event : null;
159:         $albumId = $options->album ? $options->album : null;
160: 
161:         // Get remote file list
162:         switch ($options->target)
163:         {
164:             case 'album':
165: 
166:                 // Create the album
167:                 if (! $albumId) {
168:                     list($eventId, $albumId) = $this->createAlbum($options, basename($dir));
169:                 }
170: 
171:                 $this->logger->addDebug('Fetching album photos', [$albumId]);
172:                 if ($baseOptions->preview) {
173:                     $remoteFiles = new \ArrayObject();
174:                 } else {
175:                     $remoteFiles = new ResultPager(function ($page) use ($albumId) {
176:                         $response = $this->api->getAlbumPhotos($albumId, $page + 1);
177:                         return [
178:                             (int) $response['total_pages'],
179:                             $response['photos']
180:                         ];
181:                     });
182:                 }
183:                 break;
184: 
185:             case 'event':
186: 
187:                 // Create the event
188:                 if (! $eventId) {
189:                     $eventId = $this->createEvent($options, basename($dir));
190:                 }
191: 
192:                 $this->logger->addDebug('Fetching event photos', [$eventId]);
193:                 if ($baseOptions->preview) {
194:                     $remoteFiles = new \ArrayObject();
195:                 } else {
196:                     $remoteFiles = new ResultPager(function ($page) use ($eventId) {
197:                         $response = $this->api->getEventPhotos($eventId, $page + 1);
198:                         return [
199:                             (int) $response['total_pages'],
200:                             $response['photos']
201:                         ];
202:                     });
203:                 }
204:                 break;
205:         }
206: 
207:         // Turn the response into an array of stringifiable objects so we can compare file names
208:         $remoteFiles = array_map([new ShootProofFile(), 'arrayFactory'], $remoteFiles->getArrayCopy());
209: 
210:         // Compute files to add, remove, and replace
211:         $calculator = new FileSetCalculator($localFiles, $remoteFiles, $options->replace);
212: 
213:         foreach ($calculator->add() as $file) {
214:             $filePath = $dir . DIRECTORY_SEPARATOR . $file;
215:             $this->logger->addNotice('Uploading new file to ShootProof', [$filePath]);
216:             if (! $baseOptions->preview) {
217:                 $this->uploadFile($filePath, $eventId, $albumId, $options->retryLimit);
218:             }
219:         }
220: 
221:         foreach ($calculator->remove() as $file) {
222:             $this->logger->addNotice('Deleting remote file', [$file->getId(), (string) $file]);
223:             if (! $baseOptions->preview) {
224:                 $this->deleteFile($file->getId(), $options->retryLimit);
225:             }
226:         }
227: 
228:         foreach ($calculator->replace() as $file) {
229:             $filePath = $dir . DIRECTORY_SEPARATOR . $file;
230:             $this->logger->addNotice('Replacing remote file', [$file->getId(), $filePath]);
231:             if (! $baseOptions->preview) {
232:                 $this->updateFile($filePath, $file->getId(), $options->retryLimit);
233:             }
234:         }
235: 
236:         // Write ShootProof metadata to the directory
237:         try {
238:             $writer = new ConfigWriter([
239:                 'target' => $options->target,
240:                 'event' => $eventId,
241:                 'album' => $albumId,
242:             ]);
243:             if (! $baseOptions->preview) {
244:                 $writer->write($configPath);
245:             }
246:             $this->logger->addDebug('ShootProof settings file saved', [$configPath]);
247:         } catch (\InvalidArgumentException $e) {
248:             $this->logger->addWarning('ShootProof settings file is unwritable', [$configPath]);
249:         }
250:         catch (\RuntimeException $e) {
251:             $this->logger->addWarning('Failed writing ShootProof settings file', [$configPath]);
252:         }
253:     }
