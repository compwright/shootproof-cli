101:     /**
102:      * Pulls (downloads) files from ShootProof to the specified local directory
103:      *
104:      * @param string $dir The directory to process
105:      * @param Options $baseOptions
106:      * @param OptionsFactory $optionsFactory
107:      * @throws ValidatorException if event or album options are not provided
108:      */
109:     protected function processDirectory($dir, Options $baseOptions, OptionsFactory $optionsFactory)
110:     {
111:         // If the directory doesn't exist, create it and any parent dirs
112:         if (! file_exists($dir)) {
113:             if ($baseOptions->preview || @mkdir($dir, 0777, true)) {
114:                 $this->logger->addNotice('Created directory', [$dir]);
115:             } else {
116:                 $this->logger->addError('Failed to create directory', [$dir]);
117:                 return;
118:             }
119:         }
120: 
121:         // Reload the options and read the directory config file
122:         $options = $optionsFactory->newInstance([], $this->getValidators(), $this->getDefaults());
123:         $configPath = new TildeExpander($dir) . '/.shootproof';
124:         $configLoader = new DotenvLoader($configPath);
125:         try {
126:             $configData = $configLoader->parse()->toArray();
127:             $options->loadOptionData($configData, false); // don't overwrite CLI data
128:             $this->logger->addDebug('ShootProof settings file found', [$configPath, $configData]);
129:         } catch (\InvalidArgumentException $e) {
130:         // ignore
131:             $this->logger->addDebug('ShootProof settings file not found', [$configPath]);
132:         }
133: 
134:         // Make sure all required options are present
135:         $options->validateAllRequired();
136:         if (! $options->event && ! $options->album) {
137:             throw new ValidatorException('Either --event or --album is required');
138:         }
139: 
140:         // Get local file list
141:         $localFiles = array_map('basename', $this->getFileList($dir));
142: 
143:         // Get remote file list
144:         switch ($options->target)
145:         {
146:             case 'album':
147:                 $this->logger->addDebug('Fetching album photos', [$options->album]);
148:                 $remoteFiles = new ResultPager(function ($page) use ($options) {
149:                     $response = $this->api->getAlbumPhotos($options->album, $page + 1);
150:                     return [
151:                         (int) $response['total_pages'],
152:                         $response['photos']
153:                     ];
154:                 });
155:                 break;
156: 
157:             case 'event':
158:                 $this->logger->addDebug('Fetching event photos', [$options->event]);
159:                 $remoteFiles = new ResultPager(function ($page) use ($options) {
160:                     $response = $this->api->getEventPhotos($options->event, $page + 1);
161:                     return [
162:                         (int) $response['total_pages'],
163:                         $response['photos']
164:                     ];
165:                 });
166:                 break;
167:         }
168: 
169:         // Turn the response into an array of stringifiable objects so we can compare file names
170:         $remoteFiles = array_map([new ShootProofFile(), 'arrayFactory'], $remoteFiles->getArrayCopy());
171: 
172:         // Compute files to add, remove, and replace
173:         $calculator = new FileSetCalculator($remoteFiles, $localFiles, $options->replace);
174: 
175:         foreach ($calculator->add() as $file) {
176:         // download
177:             $filePath = $dir . DIRECTORY_SEPARATOR . $file;
178:             $this->logger->addNotice('Downloading new file from ShootProof', [$filePath]);
179:             if (! $baseOptions->preview) {
180:                 $this->downloadFile($file->getUrl('original'), $filePath, $baseOptions);
181:             }
182:         }
183: 
184:         foreach ($calculator->remove() as $file) {
185:             $filePath = $dir . DIRECTORY_SEPARATOR . $file;
186:             $this->logger->addNotice('Deleting local file', [$filePath]);
187:             if (! $baseOptions->preview) {
188:                 @unlink($filePath);
189:             }
190:         }
191: 
192:         foreach ($calculator->replace() as $file) {
193:             $filePath = $dir . DIRECTORY_SEPARATOR . $file;
194:             $this->logger->addNotice('Replacing local file', [$filePath]);
195:             if (! $baseOptions->preview) {
196:                 @unlink($filePath);
197:                 $this->downloadFile($file->getUrl('original'), $filePath, $baseOptions);
198:             }
199:         }
200: 
201:         // Write ShootProof metadata to the directory
202:         try {
203:             $writer = new ConfigWriter([
204:                 'target' => $options->target,
205:                 $options->target => $options->{$options->target}, // event or album
206:             ]);
207:             if (! $baseOptions->preview) {
208:                 $writer->write($configPath);
209:             }
210:             $this->logger->addDebug('ShootProof settings file saved', [$configPath]);
211:         } catch (\InvalidArgumentException $e) {
212:             $this->logger->addWarning('ShootProof settings file is unwritable', [$configPath]);
213:         }
214:         catch (\RuntimeException $e) {
215:             $this->logger->addWarning('Failed writing ShootProof settings file', [$configPath]);
216:         }
217:     }

219:     /**
220:      * Downloads an individual file and stores it to local disk
221:      *
222:      * @param string $url URL of the file to download
223:      * @param string $destination Path to the location on disk where file should be saved
224:      * @param Options $options
225:      * @throws \RuntimeException if download failed and haltOnError is true
226:      */
227:     protected function downloadFile($url, $destination, Options $options)
228:     {
229:         $downloader = new FileDownloader($url);
230: 
231:         $retryLimit = $options->retryLimit ? $options->retryLimit : 1;
232:         for ($i = 1; $i <= $retryLimit; $i++) {
233:             try {
234:                 if (isset($e)) {
235:                     $this->logger->addInfo('Download failed, retrying', [$downloader->getResult('http_code')]);
236:                     unset($e);
237:                 }
238: 
239:                 $downloader->download($destination, true);
240:                 $this->logger->addDebug('Download completed', [$downloader->getResult('download_content_length')]);
241:                 return;
242:             } catch (\RuntimeException $e) {
243:                 @unlink($destination);
244: 
245:                 if ($options->haltOnError) {
246:                     throw $e;
247:                 } else {
248:                     // continue
249:                 }
250:             }
251:         }
252: 
253:         $this->logger->addError('Download failed on final attempt', [$downloader->getResult('http_code')]);
254:     }
